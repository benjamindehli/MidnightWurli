# midnight-wurli-plugin — distributable product (VST3 / AU / Standalone).
# Thin wrapper around dehli_musikk_sampler_engine (dm::ManifestPluginProcessor).

juce_add_plugin(MidnightWurli
    COMPANY_NAME            "DehliMusikk"
    BUNDLE_ID               com.dehlimusikk.midnightwurli
    VERSION                 0.1.0
    PLUGIN_MANUFACTURER_CODE Dhmk
    PLUGIN_CODE              Mwur
    FORMATS                 VST3 AU Standalone
    PRODUCT_NAME            "MidnightWurli"
    IS_SYNTH                TRUE
    NEEDS_MIDI_INPUT        TRUE
    NEEDS_MIDI_OUTPUT       FALSE
    IS_MIDI_EFFECT          FALSE
    EDITOR_WANTS_KEYBOARD_FOCUS FALSE
    COPY_PLUGIN_AFTER_BUILD TRUE
    VST3_CATEGORIES         "Instrument" "Sampler"
    AU_MAIN_TYPE            "kAudioUnitType_MusicDevice"
)

target_sources(MidnightWurli PRIVATE
    source/Plugin.cpp
)

target_compile_definitions(MidnightWurli PUBLIC
    JUCE_WEB_BROWSER=0
    JUCE_USE_CURL=0
    JUCE_VST3_CAN_REPLACE_VST2=0
)

target_link_libraries(MidnightWurli PRIVATE
    dehli_musikk_sampler_engine
    juce::juce_audio_utils
    juce::juce_recommended_config_flags
    juce::juce_recommended_warning_flags
)

# Embedded asset bundle (generated by ds-plugin-converter into assets/).
file(GLOB WURLI_ASSETS CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/samples/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/ir/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.png"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpg"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpeg")

if(WURLI_ASSETS)
    juce_add_binary_data(MidnightWurliAssets SOURCES ${WURLI_ASSETS})
    target_link_libraries(MidnightWurli PRIVATE MidnightWurliAssets)
    target_compile_definitions(MidnightWurli PRIVATE MIDNIGHTWURLI_HAS_ASSETS=1)
    list(LENGTH WURLI_ASSETS WURLI_ASSET_COUNT)
    message(STATUS "MidnightWurli: embedding ${WURLI_ASSET_COUNT} asset file(s) from assets/")
else()
    message(WARNING "MidnightWurli: no assets in assets/ — plugin will be silent. "
                    "Generate them with the dmse_convert target first.")
endif()
